% Command to include tikz and allow for easy setting up options for
% externalization and alike
\newcommand{\includetikz}[1]{%
	%\tikzexternaldisable%
	\tikzsetnextfilename{#1}%
	\input{#1.tikz}%
	%\tikzexternalenable%
}

% inserts 'g'reek 'b'old 'f'ace symbols eg. \gbf{\alpha}.
\newcommand{\gbf}[1] {\mbox{\boldmath${#1}$\unboldmath}}

% New command for quickly transposing items
\newcommand{\transp}[1]{\ensuremath{{#1}^{\mathrm{T}}}}

% New commands for textual things like trademark, registered, ...
\newcommand{\textR}{\textsuperscript{\textregistered}}
\newcommand{\textTM}{\textsuperscript{\texttrademark}}

% Allow typesetting something like i = 1, ..., 10
\newcommand{\numrangemath}[2]{\ensuremath{#1 = 1, \ldots, #2}}
% Allow typesetting something like i = 1, 3, ..., 9
\newcommand{\numrangemathdelta}[3]{\ensuremath{#1 = 1, #3, \ldots, #2}}
